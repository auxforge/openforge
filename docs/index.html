Get started in 2 minutes

From install to running your first agent.

Install

npm install -g openforge

Create an agent

openforge create my-agent "A helpful assistant"

This creates a .agent.json manifest in ~/.openforge/agents/

Run it

OPENAI_API_KEY=sk-xxx openforge run my-agent "hello"

Works with any provider — swap OpenAI for Anthropic, Gemini, or local Ollama.

Or use it in code

Full programmatic control with streaming support.

const { AgentBuilder } = require('openforge');

const builder = new AgentBuilder({
  aiProvider: async (messages, model, tools) => {
    // Use any AI provider you want
    const { OpenAI } = require('openai');
    return await new OpenAI()
      .chat.completions.create({ model, messages, tools });
  },
});

await builder.initialize();

// Connect an MCP server
await builder.addMcpServer('calendar', {
  transport: 'stdio',
  command: 'npx',
  args: ['-y', '@anthropic/mcp-google-calendar'],
});

// Run with real-time streaming
const { runtime, resultPromise } = await builder.runAgentStreaming(
  'my-agent', 'Check my calendar for tomorrow'
);

runtime.on('tool_call', (d) => console.log(`🔧 ${d.tool}`));
const result = await resultPromise;
